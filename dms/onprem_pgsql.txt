sudo yum install -y postgresql17
sudo yum install -y https://download.postgresql.org/pub/repos/yum/17/redhat/rhel-8-x86_64/pglogical_17-2.4.5-1PGDG.rhel8.x86_64.rpm

sudo cp /usr/pgsql-17/lib/* /usr/lib64/pgsql/
sudo cp /usr/pgsql-17/share/* /usr/share/pgsql/ -r
sudo cp /usr/pgsql-17/bin/* /usr/bin/

sudo postgresql-setup --initdb
sudo vim /var/lib/pgsql/data/postgresql.conf
---
shared_preload_libraries = 'pglogical'
max_wal_senders = 10
max_replication_slots = 10
wal_level = logical
listen_addresses = '*'
---

sudo vim /var/lib/pgsql/data/pg_hba.conf
---
host    all             all             0.0.0.0/0               md5
---

sudo systemctl restart postgresql

sudo -u postgres psql
---
create user migrate with password 'migrateadmin!!';
alter user migrate with superuser;

create database test;
grant all on database test to migrate;

\c test
create extension pglogical;
---
